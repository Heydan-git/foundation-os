#!/usr/bin/env bash
# Audit memoires : detect non-utilisees > 30j (exit 0 toujours, rapport only)
set -euo pipefail

TODAY=$(date +%s)

cat > scripts/tier-contradiction-check.sh <<'SHELL'
#!/usr/bin/env bash
# Detect phrases >= 40 chars dupliquees entre 5 tiers memoire
# Exit 0 toujours (rapport only, pas bloquant)
set -euo pipefail
REPO_ROOT="$(git rev-parse --show-toplevel)"
MEMDIR="$HOME/.claude/projects/-Users-kevinnoel-foundation-os/memory"

TIER_1="$REPO_ROOT/CLAUDE.md"
TIER_2="$REPO_ROOT/CONTEXT.md"
TIER_3_DIR="$MEMDIR"
TIER_4_DIR="$REPO_ROOT/docs/core"
TIER_5_DIR="$REPO_ROOT/wiki/meta"

DUP_COUNT=0
TMPFILE=$(mktemp)
trap "rm -f $TMPFILE" EXIT

# Extraire phrases significatives (> 40 chars, pas code block, pas table)
extract_sentences() {
  local file="$1"
  grep -vE '^(#|>|\||-|\*|\s*$|`|\s{4})' "$file" 2>/dev/null | \
    tr '.' '\n' | \
    awk 'length > 40 && length < 300'
}

# Comparer TIER_1 vs TIER_3
extract_sentences "$TIER_1" | while IFS= read -r SENT; do
  [ -z "$SENT" ] && continue
  MATCHES=$(grep -lF "$SENT" "$TIER_3_DIR"/*.md 2>/dev/null || true)
  if [ -n "$MATCHES" ]; then
    echo "[DUP CLAUDE↔memory] $SENT" >> "$TMPFILE"
    echo "  → $MATCHES" >> "$TMPFILE"
  fi
done

# TIER_2 vs TIER_5
extract_sentences "$TIER_2" | while IFS= read -r SENT; do
  [ -z "$SENT" ] && continue
  MATCHES=$(grep -lF "$SENT" "$TIER_5_DIR"/*.md 2>/dev/null || true)
  if [ -n "$MATCHES" ]; then
    echo "[DUP CONTEXT↔wiki/meta] $SENT" >> "$TMPFILE"
    echo "  → $MATCHES" >> "$TMPFILE"
  fi
done

# TIER_1 vs TIER_4
extract_sentences "$TIER_1" | while IFS= read -r SENT; do
  [ -z "$SENT" ] && continue
  MATCHES=$(grep -lF "$SENT" "$TIER_4_DIR"/*.md 2>/dev/null || true)
  if [ -n "$MATCHES" ]; then
    echo "[DUP CLAUDE↔docs/core] $SENT" >> "$TMPFILE"
    echo "  → $MATCHES" >> "$TMPFILE"
  fi
done

# TIER_2 vs TIER_4
extract_sentences "$TIER_2" | while IFS= read -r SENT; do
  [ -z "$SENT" ] && continue
  MATCHES=$(grep -lF "$SENT" "$TIER_4_DIR"/*.md 2>/dev/null || true)
  if [ -n "$MATCHES" ]; then
    echo "[DUP CONTEXT↔docs/core] $SENT" >> "$TMPFILE"
    echo "  → $MATCHES" >> "$TMPFILE"
  fi
done

# Afficher + compter
if [ -s "$TMPFILE" ]; then
  cat "$TMPFILE"
  DUP_COUNT=$(grep -c "^\[DUP" "$TMPFILE" || echo 0)
  echo "---"
  echo "Total duplications detectees : $DUP_COUNT"
else

fi
